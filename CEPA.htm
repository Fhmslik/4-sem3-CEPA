        // --- 第一關初始化 ---

        function initS1() {

            const container = document.getElementById('item-container');

            container.innerHTML = '';

            s1_items.forEach(item => {

                const card = document.createElement('div');

                card.className = 'item-card bg-white p-5 rounded-3xl shadow-md border-b-4 border-gray-200 flex flex-col items-center w-36';

                    <div class="mt-2 text-[10px] ${item.isHK ? 'bg-red-500 text-white' : 'bg-gray-200 text-gray-500'} px-2 py-0.5 rounded-full font-black uppercase">

                        ${item.isHK ? 'Made in HK' : 'Imported'}

                    </div>

                `;

                card.addEventListener('dragstart', (e) => {

                    e.dataTransfer.setData('application/json', JSON.stringify(item));

                    setTimeout(() => card.classList.add('opacity-0'), 0);

                });

                card.addEventListener('dragend', () => card.classList.remove('opacity-0'));

                container.appendChild(card);

            });

        }

 

        // --- 拖放邏輯 ---

        const dropZone = document.getElementById('drop-zone');

        dropZone.addEventListener('dragover', (e) => { e.preventDefault(); dropZone.classList.add('active'); });

        dropZone.addEventListener('drop', (e) => {

            e.preventDefault();

            dropZone.classList.remove('active');

            const data = JSON.parse(e.dataTransfer.getData('application/json'));

           

            if(data.isHK) {

                score += 100;

                s1_correct++;

                showFeedback(data.desc, "bg-emerald-600 text-white");

                document.getElementById(`item-${data.id}`).remove();

                updateUI();

                if(s1_correct === s1_target) completeS1();

            } else {

                showFeedback(data.desc, "bg-rose-600 text-white");

                const card = document.getElementById(`item-${data.id}`);

                card.classList.add('animate-shake');

                setTimeout(() => card.classList.remove('animate-shake'), 500);

            }

        });

        function loadQuiz() {

            if(s2_index >= s2_questions.length) {

                document.getElementById('quiz-container').classList.add('hidden-stage');

                document.getElementById('game-over').classList.remove('hidden-stage');

                return;

            }

            const q = s2_questions[s2_index];

            document.getElementById('quiz-progress-text').innerText = `問題 ${s2_index + 1} / ${s2_questions.length}`;

            document.getElementById('char-icon').innerText = q.icon;

            document.getElementById('quiz-question').innerText = q.q;

           

            const optionsBox = document.getElementById('quiz-options');

            optionsBox.innerHTML = '';

            q.options.forEach((opt, idx) => {

                const btn = document.createElement('button');

                btn.className = 'ans-btn bg-white hover:bg-blue-50 text-blue-900 p-6 rounded-3xl font-bold text-lg shadow-md border-2 border-blue-100 transition-all text-left';

                btn.innerHTML = `<span class="inline-block bg-blue-100 text-blue-600 w-8 h-8 rounded-full text-center leading-8 mr-3 font-bold">${idx + 1}</span> ${opt.text}`;

                btn.onclick = () => {

                    if(opt.correct) {

                        score += 70; // 調整分數比例使總分趨向 1000

                        showFeedback("答對了！" + opt.info, "bg-emerald-600 text-white");

                    } else {

                        showFeedback("答錯囉。" + opt.info, "bg-orange-600 text-white");

                    }

                    s2_index++;

                    updateUI();

                    setTimeout(loadQuiz, 2500);

                };

                optionsBox.appendChild(btn);

            });

        }

 

        // --- 共用介面 ---

        function updateUI() {

            const displayScore = Math.min(score, 1000);

            document.getElementById('score-display').innerHTML = `${displayScore} <span class="text-sm text-blue-400">/ 1000</span>`;

           

            const progress = (s1_correct + s2_index) / (s1_target + s2_questions.length);

            document.getElementById('progress-bar').style.width = `${progress * 100}%`;

        }

 

        function showFeedback(msg, cls) {

            const fb = document.getElementById('feedback');

            fb.classList.remove('hidden');

            setTimeout(() => fb.classList.add('hidden'), 2300);

        }
